541 template<typename error_category = error::rsa_error>
542 std::string extract_pubkey_from_cert(const std::string& certstr, const std::string& pw, std::error_code& ec) {
543 ec.clear();
544 auto certbio = make_mem_buf_bio(certstr);
545 auto keybio = make_mem_buf_bio();
546 if (!certbio || !keybio) {
547 ec = error_category::create_mem_bio_failed;
548 return {};
549 }
550
551 std::unique_ptr<X509, decltype(&X509_free)> cert(
552 PEM_read_bio_X509(certbio.get(), nullptr, nullptr, const_cast<char*>(pw.c_str())), X509_free);
553 if (!cert) {
554 ec = error_category::cert_load_failed;
555 return {};
556 }
557 std::unique_ptr<EVP_PKEY, decltype(&EVP_PKEY_free)> key(X509_get_pubkey(cert.get()), EVP_PKEY_free);
558 if (!key) {
559 ec = error_category::get_key_failed;
560 return {};
561 }
562 if (PEM_write_bio_PUBKEY(keybio.get(), key.get()) == 0) {
563 ec = error_category::write_key_failed;
564 return {};
565 }
566
567 return write_bio_to_string<error_category>(keybio, ec);
568 }

713 template<typename error_category = error::rsa_error>
714 evp_pkey_handle load_public_key_from_string(const std::string& key, const std::string& password,
715 std::error_code& ec) {
716 ec.clear();
717 auto pubkey_bio = make_mem_buf_bio();
718 if (!pubkey_bio) {
719 ec = error_category::create_mem_bio_failed;
720 return {};
721 }
722 if (key.substr(0, 27) == "-----BEGIN CERTIFICATE-----") {
723 auto epkey = helper::extract_pubkey_from_cert<error_category>(key, password, ec);
724 if (ec) return {};
725 const int len = static_cast<int>(epkey.size());
726 if (BIO_write(pubkey_bio.get(), epkey.data(), len) != len) {
727 ec = error_category::load_key_bio_write;
728 return {};
729 }
730 } else {
731 const int len = static_cast<int>(key.size());
732 if (BIO_write(pubkey_bio.get(), key.data(), len) != len) {
733 ec = error_category::load_key_bio_write;
734 return {};
735 }
736 }
737
738 evp_pkey_handle pkey(PEM_read_bio_PUBKEY(
739 pubkey_bio.get(), nullptr, nullptr,
740 (void*)password.data())); // NOLINT(google-readability-casting) requires `const_cast`
741 if (!pkey) ec = error_category::load_key_bio_read;
742 return pkey;
743 }

1715 std::string der_to_p1363_signature(const std::string& der_signature, std::error_code& ec) const {
1716 const unsigned char* possl_signature = reinterpret_cast<const unsigned char*>(der_signature.data());
1717 std::unique_ptr<ECDSA_SIG, decltype(&ECDSA_SIG_free)> sig(
1718 d2i_ECDSA_SIG(nullptr, &possl_signature, static_cast<long>(der_signature.length())),
1719 ECDSA_SIG_free);
1720 if (!sig) {
1721 ec = error::signature_generation_error::signature_decoding_failed;
1722 return {};
1723 }
1724
1725#ifdef JWT_OPENSSL_1_0_0
1726 auto rr = helper::bn2raw(sig->r);
1727 auto rs = helper::bn2raw(sig->s);
1728#else
1729 const BIGNUM* r;
1730 const BIGNUM* s;
1731 ECDSA_SIG_get0(sig.get(), &r, &s);
1732 auto rr = helper::bn2raw(r);
1733 auto rs = helper::bn2raw(s);
1734#endif
1735 if (rr.size() > signature_length / 2 || rs.size() > signature_length / 2)
1736 throw std::logic_error("bignum size exceeded expected length");
1737 rr.insert(0, signature_length / 2 - rr.size(), '\0');
1738 rs.insert(0, signature_length / 2 - rs.size(), '\0');
1739 return rr + rs;
1740 }

1742 std::string p1363_to_der_signature(const std::string& signature, std::error_code& ec) const {
1743 ec.clear();
1744 auto r = helper::raw2bn(signature.substr(0, signature.size() / 2), ec);
1745 if (ec) return {};
1746 auto s = helper::raw2bn(signature.substr(signature.size() / 2), ec);
1747 if (ec) return {};
1748
1749 ECDSA_SIG* psig;
1750#ifdef JWT_OPENSSL_1_0_0
1751 ECDSA_SIG sig;
1752 sig.r = r.get();
1753 sig.s = s.get();
1754 psig = &sig;
1755#else
1756 std::unique_ptr<ECDSA_SIG, decltype(&ECDSA_SIG_free)> sig(ECDSA_SIG_new(), ECDSA_SIG_free);
1757 if (!sig) {
1758 ec = error::signature_verification_error::create_context_failed;
1759 return {};
1760 }
1761 ECDSA_SIG_set0(sig.get(), r.release(), s.release());
1762 psig = sig.get();
1763#endif
1764
1765 int length = i2d_ECDSA_SIG(psig, nullptr);
1766 if (length < 0) {
1767 ec = error::signature_verification_error::signature_encoding_failed;
1768 return {};
1769 }
1770 std::string der_signature(length, '\0');
1771 unsigned char* psbuffer = (unsigned char*)der_signature.data();
1772 length = i2d_ECDSA_SIG(psig, &psbuffer);
1773 if (length < 0) {
1774 ec = error::signature_verification_error::signature_encoding_failed;
1775 return {};
1776 }
1777 der_signature.resize(length);
1778 return der_signature;
1779 }

3550 const char* in_end = str.data() + str.size();

3555 wchar_t wc;

3564 wide.push_back(wc);
3565 }
3566
3567 auto& f = std::use_facet<std::ctype<wchar_t>>(loc);
3568 f.tolower(&wide[0], &wide[0] + wide.size());
3569

3571 out.reserve(wide.size());
3572 for (wchar_t wc : wide) {
